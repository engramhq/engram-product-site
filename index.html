Docker Deployments

# Dockerfile
FROM node:alpine

WORKDIR /app

COPY . .

CMD node index.mjs
// index.mjs
import http from "http";

http
  .createServer(function (req, res) {
    res.writeHead(200, { "Content-Type": "text/plain" });
    res.write("Hello World!");
    res.end();
  })
  .listen(process.env.PORT);

Deploy

$ eg deploy
$ Deployed in 723ms
